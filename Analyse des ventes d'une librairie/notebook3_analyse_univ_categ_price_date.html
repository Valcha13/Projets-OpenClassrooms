class graphic():
    def lorenz_gini(data,title,medial,mediane):
        
        n = len(data)
        #Tri des individus dans l'ordre croissant des valeurs de la variable + calcul de la somme cumulée
        #et division par la somme des observations afin que le haut de la courbe soit à 1
        lorenz = np.cumsum(np.sort(data)) / data.sum()
        lorenz = np.append([0],lorenz)

        fig = plt.figure(figsize=(6, 6))
        memo=plt.axes().axis('equal')
        xaxis=np.linspace(0,1,len(lorenz))
        
        plt.plot(xaxis,lorenz,'blue', label='M')
        memo=plt.axis([0, 1, 0, 1])

        memo=plt.plot([0, 1], [0, 1], 'r-', lw=2, label='Première bissectrice (distribution égalitaire)')
        memo=plt.xlim(0, 1)
        memo=plt.vlines(x=medial, ymin=0, ymax=0.5, color='blue', linestyle='--', linewidth=1, label='Médial')
        memo=plt.hlines(xmin=medial, xmax=0, y=0.5, color='blue', linestyle='--', linewidth=1)
        memo=plt.vlines(x=0.5, ymin=0, ymax=mediane, color='green', linestyle='--', linewidth=1, label='Médiane')
        memo=plt.hlines(xmin=0, xmax=0.5, y=mediane, color='green', linestyle='--', linewidth=1)
        memo=plt.title(title)
        memo=plt.xlabel("Distribution (%)")
        memo=plt.ylabel("Cumul (%)")
        memo=plt.legend()
        memo=plt.legend()
        memo=plt.show()
        
        #Calcul du coefficient de Gini
        # Surface sous la courbe de Lorenz.
        AUC = (lorenz.sum() -lorenz[-1]/2 -lorenz[0]/2)/n 
        # Surface entre la première bissectrice et le courbe de Lorenz
        S = 0.5 - AUC 
        gini = 2*S
        print("Indice de Gini :", round(gini,2))
